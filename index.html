function createSolarSystem(position){ let system = new THREE.Group(); system.position.copy(position); scene.add(system); // ===== SUN ===== let sun = new THREE.Mesh( new THREE.SphereGeometry(40, 32, 32), new THREE.MeshBasicMaterial({ color: 0xffcc00 }) ); system.add(sun); let light = new THREE.PointLight(0xffffff, 3, 0); system.add(light); // ===== EARTH ===== let earthOrbit = new THREE.Group(); system.add(earthOrbit); let earth = new THREE.Mesh( new THREE.SphereGeometry(12, 32, 32), new THREE.MeshStandardMaterial({ color: 0x2266ff }) ); earth.position.x = 120; // scaled distance earthOrbit.add(earth); earth.userData.population = 8000000000; // atmosphere let earthAtm = new THREE.Mesh( new THREE.SphereGeometry(14, 32, 32), new THREE.MeshBasicMaterial({ color: 0x3399ff, transparent: true, opacity: 0.2 }) ); earth.add(earthAtm); // ===== MARS ===== let marsOrbit = new THREE.Group(); system.add(marsOrbit); let mars = new THREE.Mesh( new THREE.SphereGeometry(8, 32, 32), new THREE.MeshStandardMaterial({ color: 0xff5533 }) ); mars.position.x = 200; marsOrbit.add(mars); mars.userData.population = 0; // SAVE REFERENCES FOR ANIMATION system.userData = { earthOrbit, marsOrbit }; return system; }
